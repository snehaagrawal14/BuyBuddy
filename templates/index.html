import requests

# Set your API key
api_key = "YOUR_API_KEY"

# Get recommendations for milk and eggs
response = requests.post(
    "https://api.buybuddy.com/api/recommend",
    headers={"X-API-Key": api_key},
    json={"product_ids": ["P001", "P003"]}
)

recommendations = response.json()
print(recommendations)
// Set your API key
const apiKey = "YOUR_API_KEY";

// Get recommendations for milk and eggs
fetch("https://api.buybuddy.com/api/recommend", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "X-API-Key": apiKey
  },
  body: JSON.stringify({
    product_ids: ["P001", "P003"]
  })
})
.then(response => response.json())
.then(data => console.log(data));
<?php
// Set your API key
$apiKey = "YOUR_API_KEY";

// Get recommendations for milk and eggs
$ch = curl_init("https://api.buybuddy.com/api/recommend");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode([
    "product_ids" => ["P001", "P003"]
]));
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    "Content-Type: application/json",
    "X-API-Key: " . $apiKey
]);

$response = curl_exec($ch);
curl_close($ch);

$recommendations = json_decode($response, true);
print_r($recommendations);
?>
